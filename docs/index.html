# GET – comma-separated elements
curl "http://localhost:8080/?elements=a,b,c"

# POST – JSON array body
curl -X POST http://localhost:8080/ \
  -H "Content-Type: application/json" \
  -d '["x","y","z"]'
// GET permutations
const res = await fetch('http://localhost:8080/?elements=a,b,c');
const perms = await res.json();
console.log(perms); // [["a","b","c"], ["a","c","b"], ...]

// WebSocket streaming
const ws = new WebSocket('ws://localhost:8080/ws');
ws.onopen = () => ws.send(JSON.stringify({
  action: 'start', elements: ['a','b','c','d']
}));
ws.onmessage = e => console.log(JSON.parse(e.data));
import requests, json

# GET
r = requests.get('http://localhost:8080/', params={'elements': 'a,b,c'})
perms = r.json()
print(f"{len(perms)} permutations")

# POST
r = requests.post('http://localhost:8080/',
                  json=['x', 'y', 'z'])
print(r.json())
# Run the API server
docker run -p 8080:8080 \
  -e SERVER_HOST=0.0.0.0 \
  -e MAX_ELEMENTS=8 \
  ghcr.io/baditaflorin/go_permutation_api --serve

# Or with docker-compose
cp .env.example .env   # edit DB credentials
docker-compose up

Features

⚡

Streaming

NDJSON streaming and WebSocket real-time chunked delivery — handle millions of permutations without memory blow-up.
📖

OpenAPI 3.0

Full specification with interactive Swagger UI. Try every endpoint directly from the browser.
🏛️

Architecture Decisions

Six ADRs using the Six Thinking Hats methodology — security, performance, UX, and observability trade-offs documented.
🔐

Security

Rate limiting, CORS, security headers, body-size limits, trusted-proxy enforcement, and SQL injection prevention.
🗄️

Database Input

Pull elements from PostgreSQL, MySQL, or SQLite — or upload a CSV/TSV file directly through the GUI.
🔌

WebSocket

Bidirectional protocol with start/stop control, chunk streaming, progress events, and real-time cancellation.

Endpoints

GET
/?elements=a,b,c
Returns all permutations as a JSON array. Streams NDJSON for large inputs.
POST
/
Accepts a JSON array body; returns permutations. Supports CSV/TSV multipart upload.
GET
/health
Liveness probe — returns {"status":"healthy"}.
GET
/version
Returns build version, commit SHA, and Go runtime version.
GET
/metrics
JSON or Prometheus text metrics (gated by ENABLE_METRICS).
GET
/ws
WebSocket endpoint — stream permutations in configurable chunks with real-time cancellation.
